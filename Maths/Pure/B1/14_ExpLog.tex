\dfn{Logarithms}{
	\(a^x = y \therefore \log_ay=x\)
}
\begin{tabular}{cc}
	Exponential Fact & Logarithm Fact \\ \\
	\hline \\
	\(10^3 = 1000\) & \(\log_{10}1000 = 3\) \\ \\
	\(5^4 = 625\) & \(\log_5{625} = 4\) \\ \\
	\(36^{\frac{1}{2}} = 6\) & \(log_{36}6 = \frac{1}{2}\) \\ \\
	\(2^{-3} = \frac{1}{8}\) & \(log_2{\frac{1}{8}} = 3\)
